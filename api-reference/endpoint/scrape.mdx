---
title: "Scrape"
description: "Scrape a single URL"
---

```json
curl --location 'https://api.zyvik.com/v1/scrape' \
--header 'x-zyvik-api-key: zv-<YOUR API KEY>' \
--header 'Content-Type: application/json' \
--data '{
    "url": "https://docs.zyvik.com/",
    "format": "markdown"
}'
```

## Authorization

<ParamField header="x-zyvik-api-key" type="string" required="yes">
  Header authentication `zv-<API KEY>`
</ParamField>

## Body

<ParamField body="url" type="string" required={true}>
  Single URL to scrape

  <ParamField body="format" type="string" default="markdown" required>
    Format for the output response
  </ParamField>
</ParamField>

## Response

<ResponseField name="metadata" type="Metadata Object">
  <Expandable title="properties">
    <ResponseField name="timestamp" type="string">
      Response success timestamp
    </ResponseField>

    <ResponseField name="wordCount" type="integer">
      Total words of scraped content
    </ResponseField>

    <ResponseField name="hasImages" type="boolean">
      If the content has images
    </ResponseField>

    <ResponseField name="url" type="string">
      Scraped URL
    </ResponseField>

    <ResponseField name="ogTags" type="object">
      OG tags of the URL
    </ResponseField>

    <Expandable title="properties">
      <ResponseField name="title" type="string">
        The title of the OG tag
      </ResponseField>

      <ResponseField name="type" type="string">
        Type of the content
      </ResponseField>

      <ResponseField name="url" type="string">
        The URL of the OG tag
      </ResponseField>

      <ResponseField name="description" type="string">
        The description of the OG tag
      </ResponseField>

      <ResponseField name="image" type="string">
        The image of the OG tag
      </ResponseField>

      <ResponseField name="siteName" type="string">
        The name of the OG tag
      </ResponseField>
    </Expandable>

    <ResponseField name="format" type="string">
      Output response format
    </ResponseField>

    <ResponseField name="markdown" type="string">
      Output response
    </ResponseField>
  </Expandable>
</ResponseField>
